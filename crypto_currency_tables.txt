CREATE TABLE "users" (
  "user_id" int,
  "user_name" text,
  "password" text,
  PRIMARY KEY ("user_id")
);

CREATE TABLE "currency" (
  "currency_id" varchar(5),
  "user_id" int NOT NULL REFERENCES public.users(user_id),
  "name" text,
  "price" float,
  "volume" decimal,
  PRIMARY KEY ("currency_id")
);
CREATE TABLE "amount_invested" (
  "amount_invested_id" int,
  "ticker" varchar(5) NOT NULL REFERENCES public.currency(currency_id), 
  "amount" float,
  "amount_changed" float,
  "precent_changed" float,
  PRIMARY KEY ("amount_invested_id")
);
